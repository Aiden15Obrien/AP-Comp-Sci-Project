  <!-- Text area for user input -->

  <textarea id="code" rows="10" cols="30">print("Hello, World!")</textarea><br>

  <!-- Run Code button -->

  <button onclick="runCode()">Run Code</button>

  <!-- Section for displaying output -->

  <pre id="output"></pre>

  <script>

    // This code was generated by AI (ChatGPT)

    // Output function that displays the results

    function outf(text) {

      document.getElementById("output").innerText += text;

    }

    // Run Python code using Skulpt

    function runCode() {

      var code = document.getElementById("code").value;  // Get code from textarea

      Sk.configure({output: outf});  // Configure Skulpt to use the 'outf' function for output

      Sk.misceval.asyncToPromise(function() {

        return Sk.importMainWithBody("<stdin>", false, code);  // Run the code in Skulpt

      });

    }

  </script>
